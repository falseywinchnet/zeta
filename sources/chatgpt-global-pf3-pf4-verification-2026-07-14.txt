def loose_table(max_order: int = 9):
    """P000047 loose majorant extended one order, used only at cubic scale."""
    t, z = sp.symbols("t z", positive=True)

    derivative = lambda p: sp.expand(
        -2 * t * (t - 1) * sp.diff(p, t)
    )

    reciprocal = [sp.Integer(1)]
    for _ in range(max_order):
        reciprocal.append(
            sp.expand(derivative(reciprocal[-1]) - 2 * t * reciprocal[-1])
        )

    K = [
        sum(
            abs(c) * sp.Rational(8, 5) ** power
            for (power,), c in sp.Poly(poly, t).terms()
        )
        for poly in reciprocal
    ]

    exponential = [sp.Integer(1)]
    for _ in range(max_order):
        exponential.append(
            sp.expand(
                2 * z * (sp.diff(exponential[-1], z) - exponential[-1])
            )
        )

    E = [
        sum(
            abs(c) * 12 ** power
            for (power,), c in sp.Poly(poly, z).terms()
        )
        for poly in exponential
    ]

    prefactor = [sp.Integer(2)] + [
        sp.Rational(3, 5) * value for value in K[1:]
    ]

    first = [
        16 * sum(
            sp.binomial(order, index)
            * prefactor[index]
            * E[order - index]
            for index in range(order + 1)
        )
        for order in range(max_order + 1)
    ]

    log_sums = [sp.Integer(0)] + [
        sum(
            sp.factorial(blocks - 1)
            * sp.functions.combinatorial.numbers.stirling(
                order, blocks, kind=2
            )
            for blocks in range(1, order + 1)
        )
        for order in range(1, max_order + 1)
    ]

    scale = sp.Rational(4**4, 162754)
    return {
        order: arb(str(float(log_sums[order] * 2 * first[order] * scale)))
        for order in range(1, max_order + 1)
    }

def lcm(a: int, b: int) -> int:
    return a * b // gcd(a, b)

def evaluate_box(
    name: str,
    clean: dict,
    errors: dict,
    alpha: Fraction,
    face_axis: int,
) -> bool:
    """
    Test clean - sum_d alpha^d error_d >= 0 exactly.

    Scaling every envelope constant by alpha scales a block having total
    error degree d by alpha^d.
    """
    max_error_degree = max(
        (
            degree
            for by_degree in errors.values()
            for degree in by_degree
        ),
        default=0,
    )

    numerator = alpha.numerator
    denominator = alpha.denominator
    denominator_power = denominator**max_error_degree

    negative = {}
    face_negative = 0

    monomials = set(clean) | set(errors)

    for monomial in monomials:
        residual = clean.get(monomial, 0) * denominator_power

        for degree, coefficient in errors.get(
            monomial, {}
        ).items():
            residual -= (
                coefficient
                * numerator**degree
                * denominator
                ** (max_error_degree - degree)
            )

        if residual < 0:
            negative[monomial] = residual
            if monomial[face_axis] == 0:
                face_negative += 1

    print(
        f"{name}: alpha={float(alpha):.9e} "
        f"negative={len(negative)} "
        f"face_negative={face_negative}"
    )

    for monomial, value in list(negative.items())[:5]:
        print(
            "  NEG",
            monomial,
            f"bits={abs(value).bit_length()}",
        )

    return not negative
